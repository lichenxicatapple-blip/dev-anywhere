  ```typescript
  import { describe, it, expect, beforeEach, vi } from "vitest";
  import { RelayRouter } from "#src/serve/relay-router.js";

  interface MockDeps {
    hookApprovalManager: { resolve: ReturnType<typeof vi.fn> };
    // 其他 deps 用 stub，因为本 test 只测 hook 路径
  }

  function makeRouter(overrides: Partial<MockDeps> = {}): {
    router: RelayRouter;
    deps: { hookApprovalManager: { resolve: ReturnType<typeof vi.fn> } };
  } {
    const hookApprovalManager = overrides.hookApprovalManager ?? { resolve: vi.fn(() => true) };
    const deps = {
      sessionManager: {} as never,
      workerRegistry: {} as never,
      toolApprovalManager: {} as never,
      controlHandlers: {} as never,
      relayConnection: {} as never,
      relaySend: vi.fn(),
      terminalSockets: new Map(),
      broadcastSessionList: vi.fn(),
      broadcastSessionSync: vi.fn(),
      jsonObserver: {} as never,
      hookApprovalManager,
    };
    return {
      router: new RelayRouter(deps as Parameters<typeof RelayRouter>[0] extends never ? never : never),
      deps: { hookApprovalManager },
    };
  }

  describe("RelayRouter.terminal.permission_decision", () => {
    it("routes allow to hookApprovalManager.resolve", () => {
      const { router, deps } = makeRouter();
      router.handle({
        type: "terminal.permission_decision",
        sessionId: "s1",
        requestId: "r1",
        decision: "allow",
      });
      expect(deps.hookApprovalManager.resolve).toHaveBeenCalledWith("r1", "allow");
    });

    it("routes deny to hookApprovalManager.resolve", () => {
      const { router, deps } = makeRouter();
      router.handle({
        type: "terminal.permission_decision",
        sessionId: "s1",
        requestId: "r1",
        decision: "deny",
      });
      expect(deps.hookApprovalManager.resolve).toHaveBeenCalledWith("r1", "deny");
    });

    it("swallows unknown decision without calling resolve", () => {
      const { router, deps } = makeRouter();
      router.handle({
        type: "terminal.permission_decision",
        sessionId: "s1",
        requestId: "r1",
        decision: "maybe",
      });
      expect(deps.hookApprovalManager.resolve).not.toHaveBeenCalled();
    });

    it("ignores message missing sessionId", () => {
      const { router, deps } = makeRouter();
      router.handle({
        type: "terminal.permission_decision",
        requestId: "r1",
        decision: "allow",
      });
      expect(deps.hookApprovalManager.resolve).not.toHaveBeenCalled();
    });

    it("ignores message missing requestId", () => {
      const { router, deps } = makeRouter();
      router.handle({
        type: "terminal.permission_decision",
        sessionId: "s1",
        decision: "allow",
      });
      expect(deps.hookApprovalManager.resolve).not.toHaveBeenCalled();
    });

    it("does not throw when resolve returns false (stale decision)", () => {
      const hookApprovalManager = { resolve: vi.fn(() => false) };
      const { router } = makeRouter({ hookApprovalManager });
      expect(() =>
        router.handle({
          type: "terminal.permission_decision",
          sessionId: "s1",
          requestId: "gone",
          decision: "allow",
        }),
      ).not.toThrow();
    });
  });
  ```

  ```typescript
  // client → proxy：web 侧点 Approve/Deny 把 hook 审批回写
  private onHookPermissionDecision(msg: Record<string, unknown>): void {
    const sessionId = msg.sessionId as string | undefined;
    const requestId = msg.requestId as string | undefined;
    const decision = msg.decision as string | undefined;
    if (!sessionId || !requestId) return;
    if (decision !== "allow" && decision !== "deny") return;

    const taken = this.deps.hookApprovalManager.resolve(requestId, decision);
    if (!taken) {
      serviceLogger.info(
        { sessionId, requestId },
        "Hook permission decision ignored (already taken or expired)",
      );
    }
  }
  ```
