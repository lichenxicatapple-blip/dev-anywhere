  ```typescript
  // proxy → client：Claude Code hook 事件通道（非审批）
  z.object({
    type: z.literal("terminal.hook_event"),
    sessionId: z.string().min(1),
    hookEventName: z.enum(["PreToolUse", "PostToolUse", "Stop", "UserPromptSubmit", "SessionStart"]),
    toolName: z.string().optional(),
    toolInput: z.record(z.string(), z.unknown()).optional(),
    transcriptPath: z.string().optional(),
    timestamp: z.number(),
  }),

  // proxy → client：等待远程审批
  z.object({
    type: z.literal("terminal.permission_request"),
    sessionId: z.string().min(1),
    requestId: z.string().min(1),
    toolName: z.string().optional(),
    toolInput: z.record(z.string(), z.unknown()).optional(),
    transcriptPath: z.string().optional(),
    timestamp: z.number(),
  }),

  // client → proxy：远程审批决策
  z.object({
    type: z.literal("terminal.permission_decision"),
    sessionId: z.string().min(1),
    requestId: z.string().min(1),
    decision: z.enum(["allow", "deny"]),
  }),

  // proxy → client：派生出的当前 phase（idle / thinking / tool_use / waiting / error）
  z.object({
    type: z.literal("terminal.status"),
    sessionId: z.string().min(1),
    phase: z.enum(["idle", "thinking", "tool_use", "waiting", "error"]),
    timestamp: z.number(),
  }),
  ```
