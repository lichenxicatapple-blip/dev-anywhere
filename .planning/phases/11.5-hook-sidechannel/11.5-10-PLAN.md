  ```typescript
  // PTY 模式 hook 审批的临时简版 modal。Phase 11.5 只追求功能打通，UX 细化交给 Phase 11.6。
  // 与 tool-approval-card (JSON 模式) 刻意分开：requestId 命名空间和 envelope 类型都不同（D-12）。
  import { relayClientRef } from "@/hooks/use-relay-setup";
  import { useChatStore, type HookPermissionRequest } from "@/stores/chat-store";
  import { Button } from "@/components/ui/button";

  const MAX_INPUT_CHARS = 200;

  interface HookPermissionModalProps {
    sessionId: string;
    request: HookPermissionRequest;
  }

  // 把 tool_input 格式化为字符串并截断；过长时末尾加 U+2026 (…)
  function formatInput(input: Record<string, unknown> | undefined): string {
    if (!input) return "";
    const raw = JSON.stringify(input);
    if (raw.length <= MAX_INPUT_CHARS) return raw;
    return raw.slice(0, MAX_INPUT_CHARS) + "…";
  }

  export function HookPermissionModal({ sessionId, request }: HookPermissionModalProps) {
    function send(decision: "allow" | "deny"): void {
      const relay = relayClientRef;
      if (!relay) return;
      // RelayControl flat 消息：与 Plan 04 outgoing 方向对称；走 sendControl 而非 sendEnvelope
      relay.sendControl({
        type: "terminal.permission_decision",
        sessionId,
        requestId: request.requestId,
        decision,
      });
      useChatStore.getState().clearPendingHookPermission(sessionId);
    }

    const toolName = request.toolName ?? "unknown tool";

    return (
      <div
        data-slot="hook-permission-modal"
        className="fixed bottom-4 right-4 w-80 max-w-[90vw] rounded-md border border-border bg-card p-3 shadow-lg"
      >
        <div className="text-xs text-muted-foreground mb-1">Tool approval required</div>
        <div className="text-sm font-medium mb-2">{toolName}</div>
        <pre
          data-slot="hook-permission-input"
          className="text-xs text-muted-foreground bg-muted rounded px-2 py-1 mb-3 whitespace-pre-wrap break-all"
        >
          {formatInput(request.toolInput)}
        </pre>
        <div className="flex gap-2 justify-end">
          <Button variant="outline" size="sm" onClick={() => send("deny")}>
            Deny
          </Button>
          <Button size="sm" onClick={() => send("allow")}>
            Approve
          </Button>
        </div>
      </div>
    );
  }
  ```
