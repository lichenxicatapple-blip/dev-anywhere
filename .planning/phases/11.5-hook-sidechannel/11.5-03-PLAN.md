  import { existsSync, readFileSync, mkdirSync, writeFileSync } from "node:fs";
  import { dirname } from "node:path";
  import { nanoid } from "nanoid";
  import { DAEMON_ID_PATH } from "./paths.js";

  // 加载或生成 daemon 身份 id。用于 ~/.claude/settings.json 的 hook 条目 marker 注释，
  // 让 serve 重启后仍能识别"自己写的"旧条目并清理，避免 crash-restart 堆积。
  export function loadOrCreateDaemonId(path: string = DAEMON_ID_PATH): string {
    if (existsSync(path)) {
      const existing = readFileSync(path, "utf-8").trim();
      if (existing.length > 0) return existing;
    }
    const id = nanoid(12);
    const dir = dirname(path);
    if (!existsSync(dir)) {
      mkdirSync(dir, { recursive: true });
    }
    writeFileSync(path, id, "utf-8");
    return id;
  }

  **Step 2 — 写 `apps/proxy/src/serve/settings-injector.ts`**（基于 RESEARCH.md § Pattern: settings.json atomic rename，但签名带第 3 参支持单测注入路径，prod 用 default）：

  ```typescript
  import { readFileSync, writeFileSync, renameSync, existsSync, mkdirSync } from "node:fs";
  import { dirname, join } from "node:path";
  import { homedir } from "node:os";
  import { serviceLogger } from "../common/logger.js";

  // 默认路径：生产环境下注入到用户的 Claude Code 配置
  const DEFAULT_SETTINGS_PATH = join(homedir(), ".claude", "settings.json");

  // Claude Code hook 命令模板：curl POST 到本机 hook endpoint，marker 注释用于重启清理
  function buildCurlCommand(port: number, daemonId: string): string {
    // URL 只携带 sessionId：hook_event_name 由 Claude Code 写入 body JSON，endpoint 从 body 解析
    return (
      `curl -s -X POST --max-time 86400 ` +
      `"http://127.0.0.1:${port}/hook?sessionId=$CC_ANYWHERE_SESSION_ID" ` +
      `--data-binary @- ; # cc-anywhere-hook-marker:${daemonId}`
    );
  }

  // 注入 hook 到 settings.json。先 filter 掉自己 daemonId 的旧条目（幂等），再追加。
  // atomic 写 tmp + renameSync，crash 也不会留半写文件。
  // 解析失败时重头开始写，因为用户看到的也是损坏文件，我们的任务是恢复可用状态。
  export function injectClaudeHooks(
    port: number,
    daemonId: string,
    settingsPath: string = DEFAULT_SETTINGS_PATH,
  ): void {
    let existing: Record<string, unknown> = {};
    if (existsSync(settingsPath)) {
      try {
        existing = JSON.parse(readFileSync(settingsPath, "utf-8"));
      } catch (err) {
        serviceLogger.warn(
          { err: String(err), settingsPath },
          "Failed to parse existing settings.json, starting fresh",
        );
        existing = {};
      }
    }

    const hooks = (existing.hooks ?? {}) as Record<string, unknown[]>;
    const curlCmd = buildCurlCommand(port, daemonId);
    const defaultEntry = {
      matcher: "",
      hooks: [{ type: "command", command: curlCmd, timeout: 5 }],
    };
    const permEntry = {
      matcher: "",
      hooks: [{ type: "command", command: curlCmd, timeout: 86400 }],
    };

    const eventMap: Record<string, typeof defaultEntry> = {
      PreToolUse: defaultEntry,
      PostToolUse: defaultEntry,
      PermissionRequest: permEntry,
      Stop: defaultEntry,
      UserPromptSubmit: defaultEntry,
      SessionStart: defaultEntry,
    };

    const marker = `cc-anywhere-hook-marker:${daemonId}`;
    for (const [event, entry] of Object.entries(eventMap)) {
      const arr = Array.isArray(hooks[event]) ? hooks[event] : [];
      const filtered = arr.filter((e) => !JSON.stringify(e).includes(marker));
      filtered.push(entry);
      hooks[event] = filtered;
    }

    const merged = { ...existing, hooks };
    atomicWrite(settingsPath, JSON.stringify(merged, null, 2));
  }

  // 清理本 daemonId 写入的 hook 条目；其他来源条目保留。
  // 空 event key 也一并移除，避免残留空数组。
  export function cleanupClaudeHooks(
    daemonId: string,
    settingsPath: string = DEFAULT_SETTINGS_PATH,
  ): void {
    if (!existsSync(settingsPath)) return;
    let parsed: Record<string, unknown>;
    try {
      parsed = JSON.parse(readFileSync(settingsPath, "utf-8"));
    } catch (err) {
      serviceLogger.warn({ err: String(err) }, "settings.json corrupted, cleanup skipped");
      return;
    }
    const hooks = parsed.hooks as Record<string, unknown[]> | undefined;
    if (!hooks) return;

    const marker = `cc-anywhere-hook-marker:${daemonId}`;
    let changed = false;
    for (const [event, arr] of Object.entries(hooks)) {
      if (!Array.isArray(arr)) continue;
      const filtered = arr.filter((e) => !JSON.stringify(e).includes(marker));
      if (filtered.length !== arr.length) {
        changed = true;
        if (filtered.length === 0) delete hooks[event];
        else hooks[event] = filtered;
      }
    }
    if (Object.keys(hooks).length === 0) delete parsed.hooks;

    if (changed) {
      atomicWrite(settingsPath, JSON.stringify(parsed, null, 2));
    }
  }

  // POSIX atomic：同分区 writeFileSync + renameSync 保证读方看到完整内容
  function atomicWrite(targetPath: string, data: string): void {
    const dir = dirname(targetPath);
    if (!existsSync(dir)) mkdirSync(dir, { recursive: true });
    const tmpPath = targetPath + ".tmp";
    writeFileSync(tmpPath, data, "utf-8");
    renameSync(tmpPath, targetPath);
  }
  ```
