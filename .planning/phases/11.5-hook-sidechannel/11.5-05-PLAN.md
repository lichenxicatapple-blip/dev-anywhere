  **Step 2 — 写 `apps/proxy/src/serve/hook-approval-manager.ts`**：

  ```typescript
  import { serviceLogger } from "../common/logger.js";

  type Decision = "allow" | "deny";
  type Resolver = (decision: Decision) => void;

  interface Pending {
    sessionId: string;
    resolve: Resolver;
    registeredAt: number;
    timer: ReturnType<typeof setTimeout>;
  }

  // 24h 对齐 Claude hook curl 的 --max-time 86400 秒；任一侧先到都会让 claude 收到 deny 结果。
  const APPROVAL_TIMEOUT_MS = 24 * 60 * 60 * 1000;

  // 纯数据管理器：pending 映射 requestId → (resolver + timer)。
  // 唯一副作用是日志。HTTP response 由 hook-endpoint 在 resolver 闭包里写。
  // 与 ToolApprovalManager（stdio JSON 审批）刻意独立，lifecycle 和 requestId 命名空间不同（D-12）。
  export class HookApprovalManager {
    private pending = new Map<string, Pending>();

    register(requestId: string, sessionId: string, resolve: Resolver): void {
      // 若同 requestId 已存在（极低概率，nanoid(12) 碰撞），清掉旧 timer，让新注册生效
      const existing = this.pending.get(requestId);
      if (existing) {
        clearTimeout(existing.timer);
        serviceLogger.warn(
          { requestId, sessionId },
          "Hook permission requestId collision, previous entry dropped",
        );
      }

      const timer = setTimeout(() => {
        const entry = this.pending.get(requestId);
        if (!entry) return;
        this.pending.delete(requestId);
        serviceLogger.warn(
          { requestId, sessionId },
          "Hook permission request timed out after 24h, auto-deny",
        );
        entry.resolve("deny");
      }, APPROVAL_TIMEOUT_MS);

      this.pending.set(requestId, {
        sessionId,
        resolve,
        registeredAt: Date.now(),
        timer,
      });
    }

    // 命中即 delete + resolver(decision) + clearTimeout；返回是否命中。
    // 不命中（已 resolve / drop / 过期）返回 false，调用方自行决定是否 log。
    resolve(requestId: string, decision: Decision): boolean {
      const entry = this.pending.get(requestId);
      if (!entry) return false;
      this.pending.delete(requestId);
      clearTimeout(entry.timer);
      entry.resolve(decision);
      return true;
    }

    // HTTP request 被客户端断开时清 pending；不调用 resolver，因为 response 已断写入无意义
    drop(requestId: string, reason: string): void {
      const entry = this.pending.get(requestId);
      if (!entry) return;
      this.pending.delete(requestId);
      clearTimeout(entry.timer);
      serviceLogger.info(
        { requestId, sessionId: entry.sessionId, reason },
        "Hook permission dropped without decision",
      );
    }

    // session 终止时把该 session 的所有 pending 统一 deny，因为 HTTP response 还挂着
    drainSession(sessionId: string): void {
      for (const [requestId, entry] of this.pending) {
        if (entry.sessionId !== sessionId) continue;
        this.pending.delete(requestId);
        clearTimeout(entry.timer);
        entry.resolve("deny");
        serviceLogger.info(
          { requestId, sessionId },
          "Session ended, pending hook permission auto-denied",
        );
      }
    }

    // daemon shutdown 时全部 deny 并清空
    drainAll(): void {
      for (const [requestId, entry] of this.pending) {
        clearTimeout(entry.timer);
        entry.resolve("deny");
        serviceLogger.info(
          { requestId, sessionId: entry.sessionId },
          "Daemon shutting down, pending hook permission auto-denied",
        );
      }
      this.pending.clear();
    }
  }
  ```
