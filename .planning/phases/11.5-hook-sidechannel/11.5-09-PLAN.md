  # 模拟 Claude Code hook 子进程：读环境变量 CC_ANYWHERE_SESSION_ID 和 CC_ANYWHERE_HOOK_PORT，
  # 组装 payload 并 curl POST 到 hook endpoint。
  # 第一个参数：hook_event_name（PreToolUse / PermissionRequest / ...）

    "transcript_path": "/tmp/transcript.jsonl",
    "cwd": "/tmp"
  }
  EOF
  )

  echo "$PAYLOAD" | curl -s -X POST --max-time 86400 \
    "http://127.0.0.1:${CC_ANYWHERE_HOOK_PORT}/hook?sessionId=${CC_ANYWHERE_SESSION_ID}&event=${EVENT}" \

  **Step 2 — 创建 `apps/proxy/src/__tests__/integration/hook-sidechannel.test.ts`**：

  ```typescript
  import { describe, it, expect, beforeEach, afterEach, vi } from "vitest";
  import { spawn } from "node:child_process";
  import { mkdtempSync, rmSync, readFileSync, existsSync } from "node:fs";
  import { tmpdir } from "node:os";
  import { join } from "node:path";
  import { fileURLToPath } from "node:url";
  import { startHookEndpoint } from "../../serve/hook-endpoint.js";
  import { HookApprovalManager } from "../../serve/hook-approval-manager.js";

  // 找到 fixture 脚本绝对路径
  const __dirname = fileURLToPath(new URL(".", import.meta.url));
  const FIXTURE = join(__dirname, "..", "fixtures", "fake-claude-hook.sh");

  // 等待 hook 结果文件被写出；超时 5s（curl 本身不会挂，非 PermissionRequest）
  async function waitForFile(path: string, timeoutMs = 5000): Promise<void> {
    const start = Date.now();
    while (Date.now() - start < timeoutMs) {
      if (existsSync(path)) {
        const content = readFileSync(path, "utf-8");
        if (content.length > 0) return;
      }
      await new Promise((r) => setTimeout(r, 50));
    }
    throw new Error(`Timeout waiting for file: ${path}`);
  }

  describe("hook sidechannel end-to-end", () => {
    let tmpDir: string;
    let endpoint: Awaited<ReturnType<typeof startHookEndpoint>>;
    let approvalManager: HookApprovalManager;
    let relay: { sendEnvelope: ReturnType<typeof vi.fn> };

    beforeEach(async () => {
      tmpDir = mkdtempSync(join(tmpdir(), "hook-e2e-"));
      approvalManager = new HookApprovalManager();
      relay = { sendEnvelope: vi.fn() };
      endpoint = await startHookEndpoint({
        approvalManager,
        relay: relay as never,
      });
    });

    afterEach(async () => {
      await endpoint.close();
      rmSync(tmpDir, { recursive: true, force: true });
    });

    it("propagates env through shell subprocess to curl, endpoint receives and emits envelope", async () => {
      const outFile = join(tmpDir, "out.txt");
      const child = spawn("/bin/sh", [FIXTURE, "PreToolUse", outFile], {
        env: {
          ...process.env,
          CC_ANYWHERE_SESSION_ID: "e2e-session",
          CC_ANYWHERE_HOOK_PORT: String(endpoint.port),
        },
      });

      const exitCode: number = await new Promise((resolve) => {
        child.on("exit", (code) => resolve(code ?? 1));
      });
      expect(exitCode).toBe(0);

      await waitForFile(outFile);
      expect(readFileSync(outFile, "utf-8")).toBe("ok");

      const envelopes = relay.sendEnvelope.mock.calls.map((c) => c[0]);
      const hookEv = envelopes.find((e) => e.type === "terminal.hook_event");
      expect(hookEv).toBeDefined();
      expect(hookEv.sessionId).toBe("e2e-session");
      expect(hookEv.payload.hookEventName).toBe("PreToolUse");
      expect(hookEv.payload.toolName).toBe("Bash");
      expect(hookEv.payload.toolInput).toEqual({ command: "ls" });

      const statusEv = envelopes.find((e) => e.type === "terminal.status");
      expect(statusEv?.payload.phase).toBe("tool_use");
    });

    it("PermissionRequest: curl hangs until resolve(requestId, allow)", async () => {
      const outFile = join(tmpDir, "perm.txt");
      const child = spawn("/bin/sh", [FIXTURE, "PermissionRequest", outFile], {
        env: {
          ...process.env,
          CC_ANYWHERE_SESSION_ID: "e2e-session",
          CC_ANYWHERE_HOOK_PORT: String(endpoint.port),
        },
      });

      // curl 不应在 300ms 内返回（还在挂住）
      await new Promise((r) => setTimeout(r, 300));
      expect(existsSync(outFile)).toBe(false);

      // 找到 pending requestId（从 envelope payload 拿）
      const envelopes = relay.sendEnvelope.mock.calls.map((c) => c[0]);
      const permReq = envelopes.find((e) => e.type === "terminal.permission_request");
      expect(permReq).toBeDefined();
      const requestId = permReq.payload.requestId as string;
      expect(requestId).toHaveLength(12);

      // 决策
      const taken = approvalManager.resolve(requestId, "allow");
      expect(taken).toBe(true);

      // curl 拿到响应
      const exitCode: number = await new Promise((resolve) => {
        child.on("exit", (code) => resolve(code ?? 1));
      });
      expect(exitCode).toBe(0);

      const body = readFileSync(outFile, "utf-8");
      const parsed = JSON.parse(body);
      expect(parsed.hookSpecificOutput.hookEventName).toBe("PermissionRequest");
      expect(parsed.hookSpecificOutput.decision.behavior).toBe("allow");
    });

    it("PermissionRequest: request.close triggers drop when client killed", async () => {
      const outFile = join(tmpDir, "perm-killed.txt");
      const child = spawn("/bin/sh", [FIXTURE, "PermissionRequest", outFile], {
        env: {
          ...process.env,
          CC_ANYWHERE_SESSION_ID: "e2e-session",
          CC_ANYWHERE_HOOK_PORT: String(endpoint.port),
        },
      });

      // 等 hook 注册进 pending map
      await new Promise((r) => setTimeout(r, 200));
      const envelopes = relay.sendEnvelope.mock.calls.map((c) => c[0]);
      const permReq = envelopes.find((e) => e.type === "terminal.permission_request");
      expect(permReq).toBeDefined();
      const requestId = permReq.payload.requestId as string;

      // 杀掉 curl 模拟客户端断开
      child.kill("SIGKILL");
      await new Promise((resolve) => child.on("exit", resolve));

      // 给 endpoint 一点时间处理 req.on("close")
      await new Promise((r) => setTimeout(r, 100));

      // resolve 应返回 false（pending 已被 drop）
      const taken = approvalManager.resolve(requestId, "allow");
      expect(taken).toBe(false);
    });
  });
  ```
