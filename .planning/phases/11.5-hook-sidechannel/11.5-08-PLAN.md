  **Step 2 — 在 `relayConnection.connect()` 之后插入 hook 启动 block**（定位文件里 `relayConnection.connect();` 调用，在其紧邻下方插入）：

  ```typescript
  relayConnection.connect();

  // ----- hook sidechannel lifecycle 启动（Phase 11.5）-----
  const daemonId = loadOrCreateDaemonId();
  const hookApprovalManager = new HookApprovalManager();
  const hookEndpoint = await startHookEndpoint({
    approvalManager: hookApprovalManager,
    relay: relayConnection,
  });
  // 先启 endpoint 拿到 port，再 inject settings.json + 写 hook-port 文件给 terminal 读
  try {
    injectClaudeHooks(hookEndpoint.port, daemonId);
    writeFileSync(HOOK_PORT_PATH, String(hookEndpoint.port), "utf-8");
    serviceLogger.info(
      { port: hookEndpoint.port, daemonId },
      "Hook sidechannel ready: endpoint listening and hooks injected",
    );
  } catch (err) {
    // D-15: inject 失败降级为 warn + 继续跑；endpoint 仍在跑，只是没人会触发它
    serviceLogger.warn(
      { err: String(err) },
      "Failed to inject claude hooks, continuing without hook sidechannel",
    );
  }
  ```

  **Step 4 — shutdown 流程里按序清理**（定位 shutdown 函数；通常名字是 `shutdown` 或绑 `onShutdown`）：

  在现有 shutdown 业务逻辑之后、`process.exit` 之前，插入：

  ```typescript
  // ----- hook sidechannel lifecycle 关闭（Phase 11.5）-----
  // 顺序：先 drain pending 审批，让所有挂住的 HTTP response 回 deny 给 claude；
  // 再关 HTTP server；再清 settings.json marker；最后删 hook-port 文件
  hookApprovalManager.drainAll();
  try {
    await hookEndpoint.close();
  } catch (err) {
    serviceLogger.warn({ err: String(err) }, "Hook endpoint close errored during shutdown");
  }
  try {
    cleanupClaudeHooks(daemonId);
  } catch (err) {
    // 清理失败不阻塞 shutdown，下次 serve 启动会兜底 filter 掉旧 marker
    serviceLogger.warn({ err: String(err) }, "Failed to cleanup claude hooks on shutdown");
  }
  try {
    if (existsSync(HOOK_PORT_PATH)) unlinkSync(HOOK_PORT_PATH);
  } catch {
    // ignore
  }
  ```
