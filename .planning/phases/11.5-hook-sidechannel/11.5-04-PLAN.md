    return new Promise((resolve, reject) => {

        { hostname: "127.0.0.1", port, path, method: "POST" },
        (res) => {
          let data = "";
          res.on("data", (chunk) => (data += chunk.toString()));
          res.on("end", () => resolve({ status: res.statusCode ?? 0, body: data }));
        },

      req.on("error", reject);
      req.write(body);
      req.end();

        res.on("close", () => resolve());

      req.on("error", () => resolve());
      req.write(body);

  **Step 2 — 写 `apps/proxy/src/serve/hook-endpoint.ts`**。所有发 relay 的地方一律走 `relay.sendRaw(JSON.stringify({ type, sessionId, ...flatFields, timestamp }))`，对齐 serve.ts:206-215 的 `pty_title_change` 样板：

  ```typescript
  import { createServer, type IncomingMessage, type ServerResponse, type Server } from "node:http";
  import { nanoid } from "nanoid";
  import { serviceLogger } from "../common/logger.js";

  // 依赖抽象出来便于测试；生产装配在 serve.ts。
  interface HookApprovalManagerLike {
    register(requestId: string, sessionId: string, resolve: (decision: "allow" | "deny") => void): void;
    drop(requestId: string, reason: string): void;
  }
  interface RelayLike {
    // 与 RelayConnection.sendRaw 契约一致；直发原始 JSON 字符串给 relay
    sendRaw(raw: string): void;
  }
  interface HookEndpointDeps {
    approvalManager: HookApprovalManagerLike;
    relay: RelayLike;
  }

  // Claude Code hook 事件名 → phase 的纯映射；无对应时返回 null（不广播 status）。
  export type HookPhase = "idle" | "thinking" | "tool_use" | "waiting" | "error";
  export function derivePhase(hookEventName: string): HookPhase | null {
    switch (hookEventName) {
      case "PreToolUse": return "tool_use";
      case "PostToolUse": return "thinking";
      case "Stop": return "idle";
      case "UserPromptSubmit": return "thinking";
      case "SessionStart": return "idle";
      default: return null;
    }
  }

  // 启动本机 hook HTTP endpoint，listen 127.0.0.1 + port=0（random）。
  // 三个 timeout 全 0：PermissionRequest 需要挂住 24h，任何内部默认超时都会打断 claude。
  // 返回 { port, close } 给 serve.ts 做 lifecycle 管理。
  export async function startHookEndpoint(
    deps: HookEndpointDeps,
  ): Promise<{ port: number; close: () => Promise<void> } & { _server: Server }> {
    const server = createServer((req, res) => handleRequest(req, res, deps));
    server.setTimeout(0);
    server.keepAliveTimeout = 0;
    server.requestTimeout = 0;

    const port = await new Promise<number>((resolve, reject) => {
      server.once("error", reject);
      server.listen(0, "127.0.0.1", () => {
        const addr = server.address();
        if (typeof addr === "object" && addr) resolve(addr.port);
        else reject(new Error("Failed to read hook endpoint port"));
      });
    });

    serviceLogger.info({ port }, "Hook endpoint listening on 127.0.0.1");
    return {
      port,
      _server: server,
      close: () =>
        new Promise<void>((resolve) => {
          server.close(() => resolve());
        }),
    };
  }

  function handleRequest(req: IncomingMessage, res: ServerResponse, deps: HookEndpointDeps): void {
    if (req.method !== "POST" || !req.url?.startsWith("/hook")) {
      res.writeHead(404).end();
      return;
    }
    const url = new URL(req.url, "http://127.0.0.1");
    const sessionId = url.searchParams.get("sessionId");
    if (!sessionId) {
      res.writeHead(400).end("missing sessionId");
      return;
    }

    let body = "";
    req.on("data", (chunk: Buffer) => {
      body += chunk.toString();
    });
    req.on("end", () => {
      let event: Record<string, unknown>;
      try {
        event = JSON.parse(body);
      } catch (err) {
        serviceLogger.warn({ err: String(err), bodyLen: body.length }, "Hook body parse failed");
        res.writeHead(200).end("ok");
        return;
      }

      const hookEventName = typeof event.hook_event_name === "string" ? event.hook_event_name : "";
      const toolName = typeof event.tool_name === "string" ? event.tool_name : undefined;
      const toolInput =
        event.tool_input && typeof event.tool_input === "object"
          ? (event.tool_input as Record<string, unknown>)
          : undefined;
      const transcriptPath = typeof event.transcript_path === "string" ? event.transcript_path : undefined;

      if (hookEventName === "PermissionRequest") {
        handlePermissionRequest(res, req, sessionId, { toolName, toolInput, transcriptPath }, deps);
        return;
      }

      res.writeHead(200).end("ok");

      // flat RelayControl：与 serve.ts 现有 terminal_title / pty_state 的发法一致
      deps.relay.sendRaw(
        JSON.stringify({
          type: "terminal.hook_event",
          sessionId,
          hookEventName,
          toolName,
          toolInput,
          transcriptPath,
          timestamp: Date.now(),
        }),
      );

      const phase = derivePhase(hookEventName);
      if (phase) {
        deps.relay.sendRaw(
          JSON.stringify({
            type: "terminal.status",
            sessionId,
            phase,
            timestamp: Date.now(),
          }),
        );
      }
    });
  }

  function handlePermissionRequest(
    res: ServerResponse,
    req: IncomingMessage,
    sessionId: string,
    payload: { toolName?: string; toolInput?: Record<string, unknown>; transcriptPath?: string },
    deps: HookEndpointDeps,
  ): void {
    const requestId = nanoid(12);

    deps.approvalManager.register(requestId, sessionId, (decision) => {
      const responseBody = JSON.stringify({
        hookSpecificOutput: {
          hookEventName: "PermissionRequest",
          decision: { behavior: decision },
        },
      });
      res.writeHead(200, { "Content-Type": "application/json" }).end(responseBody);
    });

    req.on("close", () => {
      deps.approvalManager.drop(requestId, "request closed");
    });

    deps.relay.sendRaw(
      JSON.stringify({
        type: "terminal.permission_request",
        sessionId,
        requestId,
        toolName: payload.toolName,
        toolInput: payload.toolInput,
        transcriptPath: payload.transcriptPath,
        timestamp: Date.now(),
      }),
    );
  }
  ```
