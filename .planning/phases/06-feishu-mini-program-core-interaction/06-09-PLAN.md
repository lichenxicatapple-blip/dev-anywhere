**summarizeToolInput helper** (extract into testable module):

Create the helper as an exported function (can live in tool-approval-card/index.tsx or a separate utils file):
```typescript
export interface ToolSummary {
  type: "edit" | "bash" | "write" | "generic";
  summary: string;
  details: unknown;
}

export function summarizeToolInput(
  toolName: string,
  input: Record<string, unknown>,
): ToolSummary {
  const name = toolName.toLowerCase();
  if (name === "edit" || name === "edit_file" || name.includes("edit")) {
    return {
      type: "edit",
      summary: (input.file_path as string) || "unknown file",
      details: { old_string: input.old_string, new_string: input.new_string },
    };
  }
  if (name === "bash" || name === "execute" || name.includes("bash")) {
    const cmd = String(input.command || "");
    return { type: "bash", summary: cmd.length > 80 ? cmd.slice(0, 80) + "..." : cmd, details: input };
  }
  if (name === "write" || name === "write_file" || name.includes("write")) {
    return {
      type: "write",
      summary: (input.file_path as string) || "unknown file",
      details: { content: String(input.content || "").slice(0, 200) },
    };
  }
  const json = JSON.stringify(input);
  return { type: "generic", summary: json.length > 80 ? json.slice(0, 80) + "..." : json, details: input };
}
```
