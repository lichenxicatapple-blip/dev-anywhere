**Responsive viewport detection** (`apps/feishu/src/hooks/use-screen-size.ts`):

Mini programs don't support CSS media queries, so use Taro API + class-based responsive switching. The hook MUST subscribe to real-time resize events via `Taro.onWindowResize` to react to orientation changes and PC window resizing, not just one-shot `getSystemInfoSync`.

```typescript
import Taro from "@tarojs/taro";
import { useState, useEffect, useCallback } from "react";

export type ScreenCategory = "phone-portrait" | "phone-landscape" | "desktop";

// Breakpoint constants
export const BREAKPOINT_LANDSCAPE = 500;  // >= 500px: phone landscape
export const BREAKPOINT_DESKTOP = 860;    // >= 860px: Feishu PC client

// Pure function for testability -- accepts deviceType for platform-aware classification
export function classifyScreen(
  windowWidth: number,
  windowHeight: number,
  deviceType: string,
): ScreenCategory {
  // PC with large window is always desktop, regardless of width
  // (but PC in sidebar mode with <500px width is still narrow)
  if (windowWidth >= BREAKPOINT_DESKTOP) return "desktop";
  if (windowWidth >= BREAKPOINT_LANDSCAPE) return "phone-landscape";
  return "phone-portrait";
}

// CSS class name for responsive styling
export function getResponsiveClass(category: ScreenCategory): string {
  switch (category) {
    case "phone-portrait": return "screen-portrait";
    case "phone-landscape": return "screen-landscape";
    case "desktop": return "screen-desktop";
  }
}

export interface ScreenInfo {
  category: ScreenCategory;
  className: string;
  windowWidth: number;
  windowHeight: number;
  deviceType: string;         // "phone" | "tablet" | "pc" | "other"
  statusBarHeight: number;    // 0 on PC, ~20-44px on phones
  safeArea: { top: number; bottom: number; left: number; right: number; width: number; height: number };
}

// React hook -- subscribes to Taro.onWindowResize for real-time tracking
export function useScreenSize(): ScreenInfo {
  const info = Taro.getSystemInfoSync();
  const [size, setSize] = useState({
    windowWidth: info.windowWidth,
    windowHeight: info.windowHeight,
    deviceType: (info as any).deviceType || "phone",
    statusBarHeight: info.statusBarHeight || 0,
    safeArea: info.safeArea || { top: 0, bottom: info.windowHeight, left: 0, right: info.windowWidth, width: info.windowWidth, height: info.windowHeight },
  });

  useEffect(() => {
    const handler = (res: { size: { windowWidth: number; windowHeight: number } }) => {
      setSize(prev => ({
        ...prev,
        windowWidth: res.size.windowWidth,
        windowHeight: res.size.windowHeight,
      }));
    };
    Taro.onWindowResize(handler);
    return () => Taro.offWindowResize(handler);
  }, []);

  const category = classifyScreen(size.windowWidth, size.windowHeight, size.deviceType);
  return {
    category,
    className: getResponsiveClass(category),
    windowWidth: size.windowWidth,
    windowHeight: size.windowHeight,
    deviceType: size.deviceType,
    statusBarHeight: size.statusBarHeight,
    safeArea: size.safeArea,
  };
}
```
