# Phase 8: Business Logic Adaptation - Research

**Researched:** 2026-04-16
**Domain:** React SPA 状态管理、路由、WebSocket 通信层迁移
**Confidence:** HIGH

## Summary

Phase 8 的核心工作是将 Feishu 小程序的非 UI 业务逻辑迁移到 apps/web，用浏览器原生 API（zustand、react-router hash 模式、原生 WebSocket、localStorage）替换所有 Taro 依赖。改动范围仅限 apps/web，不涉及 relay 或 proxy。

迁移涉及 6 个 store（app/session/chat/terminal/command/file）从 Context + useReducer 改写为 zustand，1 个 WebSocket 管理器重写（去掉 Taro 分支，加入 binary 帧处理和指数退避重连），1 个状态机适配（PhaseNav 接口溶解，直接使用 router.navigate() 和 zustand store），以及 app.tsx 的完整接线（react-router + WebSocket + 状态机 + 冷启动恢复）。

源码分析发现：chat-store、session-store、command-store、file-store 是纯 reducer 无 Taro 依赖，改写为 zustand 只需换外壳；app-store 和 terminal-store 有 Taro.getStorageSync/setStorageSync 调用需替换为 localStorage；relay-client.ts 和 ensure-binding.ts 无 Taro 依赖可直接复制；websocket.ts 需要完全重写（去掉 Taro.connectSocket 分支和 TaskLike 抽象，加入 binary 帧处理）。

**Primary recommendation:** 按依赖顺序分 3 波执行——(1) zustand stores + 路径别名 + 路由配置，(2) WebSocket 管理器 + 服务层 + 状态机，(3) app.tsx 接线 + pty-test 改造 + 验证。

### Pattern 1: Zustand Store（从 Context + useReducer 迁移）

**What:** 将 Feishu 版的 Context + useReducer 模式改写为 zustand store
**When to use:** 所有 6 个 store 的迁移

**Example:**
```typescript
// Source: Context7 /pmndrs/zustand, verified
// apps/web/src/stores/session-store.ts
import { create } from 'zustand';
import { devtools } from 'zustand/middleware';
import type { SessionInfo, HistorySession } from '@cc-anywhere/shared';

interface SessionStoreState {
  sessions: SessionInfo[];
  historySessions: HistorySession[];
  currentSessionId: string | null;
  currentSessionMode: 'pty' | 'json' | null;
  // actions 直接放在 store 中
  setSessions: (sessions: SessionInfo[]) => void;
  setCurrentSession: (sessionId: string | null, mode: 'pty' | 'json' | null) => void;
  addSession: (session: SessionInfo) => void;
  removeSession: (sessionId: string) => void;
  updateSessionState: (sessionId: string, state: SessionInfo['state']) => void;
  updateSessionName: (sessionId: string, name: string) => void;
  setHistorySessions: (sessions: HistorySession[]) => void;
}

export const useSessionStore = create<SessionStoreState>()(
  devtools(
    (set) => ({
      sessions: [],
      historySessions: [],
      currentSessionId: null,
      currentSessionMode: null,
      setSessions: (sessions) => set({ sessions }),
      setCurrentSession: (sessionId, mode) =>
        set({ currentSessionId: sessionId, currentSessionMode: mode }),
      addSession: (session) =>
        set((state) => ({ sessions: [...state.sessions, session] })),
      removeSession: (sessionId) =>
        set((state) => ({
          sessions: state.sessions.filter((s) => s.sessionId !== sessionId),
          currentSessionId: state.currentSessionId === sessionId ? null : state.currentSessionId,
          currentSessionMode: state.currentSessionId === sessionId ? null : state.currentSessionMode,
        })),
      updateSessionState: (sessionId, newState) =>
        set((state) => ({
          sessions: state.sessions.map((s) =>
            s.sessionId === sessionId ? { ...s, state: newState } : s,
          ),
        })),
      updateSessionName: (sessionId, name) =>
        set((state) => ({
          sessions: state.sessions.map((s) =>
            s.sessionId === sessionId ? { ...s, name } : s,
          ),
        })),
      setHistorySessions: (sessions) => set({ historySessions: sessions }),
    }),
    { name: 'session-store' },
  ),
);
```

### Pattern 3: WebSocket 管理器（text + binary 分发）

**What:** 统一 WebSocket 管理器同时处理 JSON text 和 binary ArrayBuffer
**When to use:** 整个应用的 WebSocket 通信层

**Example:**
```typescript
// apps/web/src/services/websocket.ts
export class WebSocketManager {
  private ws: WebSocket | null = null;
  private url = '';
  private connected = false;
  private closed = false;
  private reconnectTimer: ReturnType<typeof setTimeout> | null = null;
  private reconnectAttempt = 0;

  // text 消息处理器 (JSON)
  private messageHandlers = new Set<(data: string) => void>();
  // binary 帧订阅者，按 sessionId 过滤
  private binarySubscribers = new Map<string, Set<(data: Uint8Array) => void>>();
  // 连接状态处理器
  private statusHandlers = new Set<(connected: boolean) => void>();
  private pendingQueue: string[] = [];

  connect(url: string): void {
    // ...
    this.doConnect();
  }

  private doConnect(): void {
    const ws = new WebSocket(this.url);
    ws.binaryType = 'arraybuffer';
    this.ws = ws;

    ws.addEventListener('open', () => {
      this.connected = true;
      this.reconnectAttempt = 0;
      this.flushPendingQueue();
      this.statusHandlers.forEach((h) => h(true));
    });

    ws.addEventListener('message', (event) => {
      if (event.data instanceof ArrayBuffer) {
        this.dispatchBinary(new Uint8Array(event.data));
      } else {
        const data = event.data as string;
        this.messageHandlers.forEach((h) => h(data));
      }
    });

    ws.addEventListener('close', () => {
      this.connected = false;
      this.ws = null;
      this.statusHandlers.forEach((h) => h(false));
      if (!this.closed) this.scheduleReconnect();
    });
  }

  // D-08: 指数退避重连 1s -> 2s -> 4s -> 8s -> 30s
  private scheduleReconnect(): void {
    const delay = Math.min(1000 * Math.pow(2, this.reconnectAttempt), 30000);
    this.reconnectAttempt++;
    this.reconnectTimer = setTimeout(() => this.doConnect(), delay);
  }

  // D-06: 解析 binary 帧，按 sessionId 分发
  private dispatchBinary(view: Uint8Array): void {
    if (view.length < 2) return;
    const sidLen = view[0];
    if (view.length < 1 + sidLen) return;
    const sessionId = new TextDecoder().decode(view.subarray(1, 1 + sidLen));
    const ptyData = view.subarray(1 + sidLen);
    const subs = this.binarySubscribers.get(sessionId);
    if (subs) {
      subs.forEach((cb) => cb(ptyData));
    }
  }

  // 订阅特定 session 的 binary 数据
  subscribeBinary(sessionId: string, handler: (data: Uint8Array) => void): () => void {
    if (!this.binarySubscribers.has(sessionId)) {
      this.binarySubscribers.set(sessionId, new Set());
    }
    this.binarySubscribers.get(sessionId)!.add(handler);
    return () => {
      const subs = this.binarySubscribers.get(sessionId);
      if (subs) {
        subs.delete(handler);
        if (subs.size === 0) this.binarySubscribers.delete(sessionId);
      }
    };
  }

  // ... send(), close(), onMessage(), onStatusChange(), isConnected()
}
```

### Pattern 5: visibilitychange 检测前后台切换

**What:** 页面从后台恢复时检测 WebSocket 状态并立即重连
**When to use:** D-08 要求

**Example:**
```typescript
// 在 use-relay-setup.ts 或 app.tsx 中
document.addEventListener('visibilitychange', () => {
  if (document.visibilityState === 'visible' && ws && !ws.isConnected()) {
    ws.connect(relayUrl);
  }
});
```

### Anti-Patterns to Avoid

- **嵌套 Provider 地狱:** zustand 不需要 Provider 包装，不要在 app.tsx 中创建多层 Provider（这正是迁移到 zustand 的核心动机之一）
- **在 store 内部调用 localStorage:** D-03 明确只在特定时机手动读写，不要在 zustand middleware 中自动同步
- **循环依赖 app.tsx <-> phase-machine:** router 实例放独立模块，phase-machine import router 模块而非 app.tsx
- **重连时重建 RelayClient:** RelayClient 是有状态的（sessionSeqMap、messageHandlers），WebSocket 重连时复用同一实例，只重建底层 WebSocket 连接

## Don't Hand-Roll

| Problem | Don't Build | Use Instead | Why |
|---------|-------------|-------------|-----|
| 状态管理 | Context + useReducer + 多层 Provider | zustand create() | D-01 锁定，消除 Provider 嵌套，支持组件外访问 |
| Hash 路由 | 手动 hashchange 监听 + 路径匹配 | createHashRouter + RouterProvider | 当前 app.tsx 的手动 hash 路由必须替换 |
| WebSocket 重连 | 手写固定间隔重连 | 在 WebSocketManager 内建指数退避 | D-08 锁定策略，不用第三方重连库（如 reconnecting-websocket），因为需要同时处理 binary |
| 开发工具 | console.log 调试 state | zustand devtools middleware | 连接 Redux DevTools，零成本 |

**Key insight:** 此阶段所有"不要手写"的决策都已在 CONTEXT.md 中锁定，没有需要额外选择的领域。

## Common Pitfalls

### Pitfall 1: Router 实例循环依赖

**What goes wrong:** phase-machine.ts 需要调用 router.navigate()，如果 router 定义在 app.tsx 中，而 app.tsx 又 import phase-machine，会形成循环依赖
**Why it happens:** react-router 的 createHashRouter 需要引用页面组件，页面组件可能间接依赖状态机
**How to avoid:** 将 createHashRouter 放在独立模块 `lib/router.ts`，路由配置只引用页面组件，phase-machine 只引用 router 实例。页面组件 -> stores -> phase-machine -> router 是单向的
**Warning signs:** TypeScript 报 undefined import，Vite HMR 失效

### Pitfall 2: Zustand Store 初始化时序

**What goes wrong:** app-store 需要在创建时从 localStorage 读取 clientId，但 zustand store 是模块级别的单例，初始化发生在 import 时
**Why it happens:** zustand store 在模块加载时立即执行 create()，此时 localStorage 已可用（浏览器环境），但 vitest + jsdom 环境中 localStorage 行为可能不同
**How to avoid:** 初始化逻辑放在 store 的初始 state 函数中。vitest 的 jsdom 环境提供了 localStorage mock，初始化不会出问题。但要注意 store 的 create() 在测试文件 import 时就执行，测试之间需要 store.setState() 重置状态
**Warning signs:** 测试之间状态泄漏

### Pitfall 3: Binary 帧处理与 binaryType 设置

**What goes wrong:** WebSocket 收到 binary 数据但 event.data 是 Blob 而非 ArrayBuffer
**Why it happens:** 浏览器 WebSocket 默认 binaryType 是 'blob'，必须显式设为 'arraybuffer'
**How to avoid:** 在 WebSocket 构造后立即设置 `ws.binaryType = 'arraybuffer'`
**Warning signs:** `event.data instanceof ArrayBuffer` 始终为 false

### Pitfall 4: 指数退避重连在组件卸载后继续执行

**What goes wrong:** WebSocket 管理器的 reconnect timer 在页面卸载后继续触发
**Why it happens:** WebSocketManager 是模块级单例，不随组件生命周期销毁
**How to avoid:** close() 方法必须清除 reconnect timer 并设 closed 标志阻止后续重连。app.tsx 的 useEffect cleanup 中调用 close()
**Warning signs:** 控制台不断打印重连日志

### Pitfall 5: phase-machine handleRelayMessage 中的异步竞态

**What goes wrong:** ensureBinding 是 async 的，多个 proxy_list_response 可能并发执行冷启动逻辑
**Why it happens:** relay 在短时间内推送多次 proxy_list_response（如 proxy 重连时）
**How to avoid:** 用 coldStartDone 标志位防止重复执行。zustand 的 getState() 始终返回最新状态，不存在 React 的 stale closure 问题
**Warning signs:** 多次路由跳转、状态错乱

### Pitfall 6: pty-test 改造后 binary 订阅泄漏

**What goes wrong:** pty-test 页面切换 sessionId 后，旧 sessionId 的 binary 订阅未清理
**Why it happens:** 没有调用 subscribeBinary 返回的 unsubscribe 函数
**How to avoid:** 每次 sessionId 变化时先 unsubscribe 旧的再 subscribe 新的，useEffect cleanup 中 unsubscribe
**Warning signs:** 切换 session 后终端显示混合数据

### App Store 带 localStorage 持久化

```typescript
// Source: Context7 /pmndrs/zustand, verified
// apps/web/src/stores/app-store.ts
import { create } from 'zustand';
import { devtools } from 'zustand/middleware';
import type { ProxyInfo } from '@cc-anywhere/shared';

export type AppPhase =
  | 'connecting'
  | 'registering'
  | 'reconnecting'
  | 'proxy_selecting'
  | 'session_browsing'
  | 'chatting';

interface AppStoreState {
  phase: AppPhase;
  phaseBeforeDisconnect: AppPhase | null;
  connected: boolean;
  proxyOnline: boolean;
  selectedProxyId: string | null;
  selectedProxyName: string | null;
  proxies: ProxyInfo[];
  clientId: string;
  relayUrl: string;

  // actions
  setConnected: (connected: boolean) => void;
  setProxy: (proxyId: string | null, proxyName: string | null) => void;
  setProxyOnline: (online: boolean) => void;
  setRelayUrl: (url: string) => void;
  setPhase: (phase: AppPhase) => void;
  setProxies: (proxies: ProxyInfo[]) => void;
  transitionToPhase: (next: AppPhase) => void;
}

function loadClientId(): string {
  const stored = localStorage.getItem('cc_clientId');
  if (stored) return stored;
  const id = Date.now().toString(36) + Math.random().toString(36).slice(2);
  localStorage.setItem('cc_clientId', id);
  return id;
}

export const useAppStore = create<AppStoreState>()(
  devtools(
    (set, get) => ({
      phase: 'connecting',
      phaseBeforeDisconnect: null,
      connected: false,
      proxyOnline: false,
      selectedProxyId: null,
      selectedProxyName: null,
      proxies: [],
      clientId: loadClientId(),
      relayUrl: '',

      setConnected: (connected) => set({ connected }),
      setProxy: (proxyId, proxyName) => set({ selectedProxyId: proxyId, selectedProxyName: proxyName }),
      setProxyOnline: (online) => set({ proxyOnline: online }),
      setRelayUrl: (url) => set({ relayUrl: url }),
      setPhase: (phase) => {
        const prev = get().phase;
        const phaseBeforeDisconnect = phase === 'reconnecting' ? prev : get().phaseBeforeDisconnect;
        set({ phase, phaseBeforeDisconnect });
      },
      setProxies: (proxies) => set({ proxies }),
      transitionToPhase: (next) => {
        const prev = get().phase;
        // D-03: 特定时机清理 localStorage
        if (next === 'proxy_selecting') {
          localStorage.removeItem('cc_proxyId');
          localStorage.removeItem('cc_sessionId');
        }
        if (next === 'session_browsing' && prev === 'chatting') {
          localStorage.removeItem('cc_sessionId');
        }
        const phaseBeforeDisconnect = next === 'reconnecting' ? prev : get().phaseBeforeDisconnect;
        set({ phase: next, phaseBeforeDisconnect });
      },
    }),
    { name: 'app-store' },
  ),
);
```

// apps/web/src/stores/toast-store.ts
import { create } from 'zustand';

interface Toast {
  id: number;
  message: string;
}

interface ToastStoreState {
  toasts: Toast[];
  showToast: (message: string) => void;
  removeToast: (id: number) => void;
}

let nextId = 0;

export const useToastStore = create<ToastStoreState>()((set) => ({
  toasts: [],
  showToast: (message) => {
    const id = nextId++;
    set((state) => ({ toasts: [...state.toasts, { id, message }] }));
    setTimeout(() => {
      set((state) => ({ toasts: state.toasts.filter((t) => t.id !== id) }));
    }, 3000);
  },
  removeToast: (id) =>
    set((state) => ({ toasts: state.toasts.filter((t) => t.id !== id) })),
}));
