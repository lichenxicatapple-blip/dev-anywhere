    **Edit B — apps/web/src/components/proxy/proxy-switcher.tsx (new):**
    ```tsx
    // ProxySwitcher dual-layout 组件
    // layout=page: 移动端全屏列表（覆盖当前 proxy-select.tsx 逻辑）
    // layout=dropdown: 桌面侧栏顶部 Popover，选中后不跳路由，仅更新 app-store
    import { useNavigate } from "react-router";
    import { useAppStore } from "@/stores/app-store";
    import { relayClientRef } from "@/services/ensure-binding";
    import { showErrorToast } from "@/components/toast";
    import { Button } from "@/components/ui/button";
    import {
      Popover,
      PopoverTrigger,
      PopoverContent,
    } from "@/components/ui/popover";
    import { ProxyStatusDot } from "./proxy-status-dot";
    import { EmptyState } from "@/components/shell/empty-state";

    interface ProxySwitcherProps {
      layout: "page" | "dropdown";
    }

    export function ProxySwitcher({ layout }: ProxySwitcherProps) {
      const proxies = useAppStore((s) => s.proxies);
      const selectedProxyId = useAppStore((s) => s.selectedProxyId);
      const navigate = useNavigate();

      async function handleSelect(proxyId: string, proxyName: string | null | undefined) {
        const relay = relayClientRef.current;
        if (!relay) {
          showErrorToast("Relay client not available");
          return;
        }
        const result = await relay.selectProxy(proxyId);
        if (!result.success) {
          showErrorToast(`选择 Proxy 失败: ${result.error ?? "unknown"}`);
          return;
        }
        localStorage.setItem("cc_proxyId", proxyId);
        useAppStore.getState().setProxy(proxyId, proxyName ?? null);
        useAppStore.getState().setProxyOnline(true);
        useAppStore.getState().transitionToPhase("session_browsing");
        if (layout === "page") {
          navigate("/sessions");
        }
      }

      if (layout === "page") {
        if (proxies.length === 0) {
          return <EmptyState variant="no-proxy" />;
        }
        return (
          <div className="flex flex-col gap-2 p-4 h-full overflow-auto">
            <h2 className="text-lg font-semibold mb-2">选择一个 Proxy</h2>
            <ul role="list" className="flex flex-col gap-2">
              {proxies.map((p) => (
                <li key={p.proxyId}>
                  <button
                    type="button"
                    data-slot="proxy-item"
                    data-proxy-id={p.proxyId}
                    onClick={() => handleSelect(p.proxyId, p.name)}
                    className="w-full flex items-center gap-3 px-3 h-11 min-h-[44px] rounded-md border border-border bg-card hover:bg-accent transition-colors text-left"
                    aria-pressed={selectedProxyId === p.proxyId}
                  >
                    <ProxyStatusDot status={p.online ? "online" : "offline"} />
                    <span className="text-sm font-normal flex-1">{p.name ?? p.proxyId}</span>
                    {!p.online && <span className="text-xs text-muted-foreground">离线</span>}
                  </button>
                </li>
              ))}
            </ul>
          </div>
        );
      }

      // layout === "dropdown"
      const currentProxy = proxies.find((p) => p.proxyId === selectedProxyId);
      return (
        <Popover>
          <PopoverTrigger asChild>
            <Button
              variant="ghost"
              size="sm"
              className="w-full justify-start gap-2 h-9"
              data-slot="proxy-switcher-trigger"
            >
              <ProxyStatusDot status={currentProxy?.online ? "online" : "offline"} />
              <span className="text-sm font-normal truncate flex-1 text-left">
                {currentProxy?.name ?? currentProxy?.proxyId ?? "未选择"}
              </span>
            </Button>
          </PopoverTrigger>
          <PopoverContent align="start" className="w-[260px] p-1">
            {proxies.length === 0 ? (
              <div className="text-xs text-muted-foreground p-2">没有可用 Proxy</div>
            ) : (
              <ul role="list" className="flex flex-col">
                {proxies.map((p) => (
                  <li key={p.proxyId}>
                    <button
                      type="button"
                      data-slot="proxy-item"
                      data-proxy-id={p.proxyId}
                      onClick={() => handleSelect(p.proxyId, p.name)}
                      className="w-full flex items-center gap-2 px-2 h-9 rounded-md hover:bg-accent transition-colors text-left"
                      aria-pressed={selectedProxyId === p.proxyId}
                    >
                      <ProxyStatusDot status={p.online ? "online" : "offline"} />
                      <span className="text-sm font-normal flex-1 truncate">{p.name ?? p.proxyId}</span>
                    </button>
                  </li>
                ))}
              </ul>
            )}
          </PopoverContent>
        </Popover>
      );
    }
    ```

    Per D-10 (shared business component) and D-04 (FRONT-04). Touch-target height 44px mobile (min-h-[44px]) per UI-SPEC Spacing iOS HIG rule; 36px (h-9) desktop.

| T-10-02-02 | Information Disclosure | Proxy names displayed in sidebar + popover | accept | Names are already user-visible in the proxy list — no incremental exposure |
| T-10-02-03 | Denial of Service | handleSelect shows error toast on failure | mitigate | Explicit error handling with showErrorToast (RESEARCH §2.6) — relay errors surfaced to user without crashing |
| T-10-02-04 | Tampering | localStorage cc_proxyId manipulation | accept | On load, ensure-binding validates via selectProxy; invalid id triggers error path to re-select |
</threat_model>
