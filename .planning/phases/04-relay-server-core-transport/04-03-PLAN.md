    ```typescript
    import WebSocket from "ws";
    import { readFileSync, writeFileSync, mkdirSync, existsSync } from "node:fs";
    import { join } from "node:path";
    import { nanoid } from "nanoid";
    import { EventEmitter } from "node:events";
    import type { Logger } from "pino";
    import type { MessageEnvelope } from "@cc-anywhere/shared";

    // proxyId 配置文件路径
    const PROXY_ID_PATH = join(
      process.env.HOME ?? process.env.USERPROFILE ?? ".",
      ".cc-anywhere",
      "proxy-id",
    );

    export class RelayConnection extends EventEmitter {
      private ws: WebSocket | null = null;
      private proxyId: string;
      private logger: Logger;
      private relayUrl: string;

      constructor(relayUrl: string, logger: Logger);

      // 读取或生成 proxyId
      private loadOrCreateProxyId(): string;

      // 连接到 relay server
      connect(): void;

      // 发送 MessageEnvelope 到 relay
      send(envelope: MessageEnvelope): void;

      // 关闭连接
      close(): void;

      // 获取 proxyId
      getProxyId(): string;
    }
    ```

    Implementation details:
    - `loadOrCreateProxyId()`: Read from `PROXY_ID_PATH`. If file exists, return its content (trimmed). If not, generate `nanoid(21)`, write to file (creating directory if needed), return it.
    - `connect()`: Create `new WebSocket(this.relayUrl + "/proxy")`. On `open`, send `JSON.stringify({ type: "proxy_register", proxyId: this.proxyId })`. On `message`, emit `"message"` event with the parsed data. On `close`, log disconnection. On `error`, log error. Per D-11, no reconnection logic in Phase 4.
    - `send(envelope)`: If `ws` is open (`readyState === WebSocket.OPEN`), call `ws.send(JSON.stringify(envelope))`. Otherwise log warning.
    - `close()`: If ws exists, call `ws.close()`.

    The relayUrl comes from environment variable `RELAY_URL` or a default. The constructor accepts it as parameter so serve.ts can configure it.

    ```bash
    #!/usr/bin/env bash
    set -euo pipefail

    # CC Anywhere Relay Server 部署脚本
    # 用法: ./deploy.sh <ssh-host> [domain]
    # 示例: ./deploy.sh root@1.2.3.4 relay.example.com

    SSH_HOST="${1:?Usage: ./deploy.sh <ssh-host> [domain]}"
    DOMAIN="${2:-}"
    SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
    PROJECT_ROOT="$(cd "$SCRIPT_DIR/../.." && pwd)"

    echo "=== CC Anywhere Relay Deployment ==="
    echo "Target: $SSH_HOST"
    echo "Domain: ${DOMAIN:-not set}"
    echo ""

    # Step 1: 检查远程服务器上 Docker 是否安装
    echo "[1/5] Checking Docker on remote server..."
    ssh "$SSH_HOST" 'command -v docker >/dev/null 2>&1 || {
      echo "Installing Docker..."
      curl -fsSL https://get.docker.com | sh
      systemctl enable docker
      systemctl start docker
      echo "Docker installed"
    }'

    # Step 2: 确保 docker compose 可用
    ssh "$SSH_HOST" 'docker compose version >/dev/null 2>&1 || {
      echo "ERROR: docker compose plugin not available"
      exit 1
    }'

    # Step 3: SSL 证书设置
    if [ -n "$DOMAIN" ]; then
      echo "[2/5] Setting up SSL certificate for $DOMAIN..."
      ssh "$SSH_HOST" "
        command -v certbot >/dev/null 2>&1 || {
          yum install -y epel-release
          yum install -y certbot

        if [ ! -d /etc/letsencrypt/live/relay ]; then
          certbot certonly --standalone -d $DOMAIN --non-interactive --agree-tos --email admin@$DOMAIN \
            --cert-name relay
        else
          echo 'Certificate already exists'
        fi
      "
    else
      echo "[2/5] Skipping SSL setup (no domain provided)"

    REMOTE_DIR="/opt/cc-anywhere"
    ssh "$SSH_HOST" "mkdir -p $REMOTE_DIR"
    rsync -avz --delete \
      --exclude node_modules \
      --exclude dist \
      --exclude .git \
      --exclude .planning \
      --exclude .claude \
      --exclude .playwright-mcp \
      "$PROJECT_ROOT/" "$SSH_HOST:$REMOTE_DIR/"

    # Step 5: 构建并启动
    echo "[4/5] Building and starting relay server..."
    ssh "$SSH_HOST" "
      cd $REMOTE_DIR
      docker compose -f apps/relay/docker-compose.yml build --no-cache
      docker compose -f apps/relay/docker-compose.yml up -d
    "

    echo "=== Deployment complete ==="

    echo "Health: curl http://<server-ip>:3100/health"
    ```
